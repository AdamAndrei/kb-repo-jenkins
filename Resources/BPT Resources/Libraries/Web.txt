		Case "click"
			If  iX <> "" AND iY <> "" Then

				objWebButton.highlight
				objWebButton.Click
''				wait 2
			End If
'			Reporter.ReportEvent micPass, "Click on web button", "Successfully clicked WebButton [ "& objWebButton.toString &" ]"

Public Function Fn_Web_UI_ChooseFile_Operations(sAction,sFilePath)
	Err.Clear
	'--------------------------------------------------------------------------------------------------------------------------------
	'Initially set function return value as False
	'--------------------------------------------------------------------------------------------------------------------------------
	Fn_Web_UI_ChooseFile_Operations = False
	'--------------------------------------------------------------------------------------------------------------------------------
	'Variable Declaration
	'--------------------------------------------------------------------------------------------------------------------------------
	Dim obj_AWCTeamcenterHome,objWscript
	Dim sTempValue,sTemp
	'-------------------------------------------------------------------------------------------------------------------------------
	'Get AWC window object from xml
	'--------------------------------------------------------------------------------------------------------------------------------
	Set obj_AWCTeamcenterHome=Eval(SearchAndLoadResourceByName("ActiveWorkspace2406_OR.xml").GetValue("wpage_AWCTeamcenterHome"))
	Set objWscript=CreateObject("wscript.shell")
	
	'--------------------------------------------------------------------------------------------------------------------------------
	Select Case lcase(sAction)
		'--------------------------------------------------------------------------------------------------------------------------------
		'Case to choose file from local drive
		Case "choosefile","choosetemplate"
			'--------------------------------------------------------------------------------------------------------------------------------
			If WaitUntilExist(obj_AWCTeamcenterHome.WebFile("wfile_ChooseFile"), 60,1) Then
				obj_AWCTeamcenterHome.WebFile("wfile_ChooseFile").Highlight
				wait 2
				obj_AWCTeamcenterHome.WebFile("wfile_ChooseFile").set sFilePath
				wait 6
				sTemp=obj_AWCTeamcenterHome.WebFile("wfile_ChooseFile").GetROProperty("title")
				'--------------------------------------------------------------------------------------------------------------------------------	
				If instr( sFilePath,sTemp)< 0 Then
					obj_AWCTeamcenterHome.WebFile("wfile_ChooseFile").set sFilePath
					wait 6
				End  IF
				'--------------------------------------------------------------------------------------------------------------------------------
				sTemp=obj_AWCTeamcenterHome.WebFile("wfile_ChooseFile").GetROProperty("title")
				If instr( sFilePath,sTemp)> 0 Then
					Reporter.ReportEvent micPass, "Upload " & sFilePath & " File ", "Successfully  uploaded " & sFilePath & " File "
					Fn_Web_UI_ChooseFile_Operations=True
				Else
					Reporter.ReportEvent micFail, "Upload " & sFilePath & " File ", "Falied to upload file " & sFilePath
					ExitComponent	
				End If
				'--------------------------------------------------------------------------------------------------------------------------------
			End  If
			'--------------------------------------------------------------------------------------------------------------------------------
	End  Select	
'--------------------------------------------------------------------------------------------------------------------------------
'Report any unexpected runtime error
'--------------------------------------------------------------------------------------------------------------------------------
If Err.Number <> 0 Then
	Reporter.ReportEvent micFail, "Fn_Web_UI_ChooseFile_Operations", "Fail To perform operation [ " & Cstr(sAction) & " ] due to error number [ " & Cstr(Err.Number) & " ] with error description [ " & Cstr(Err.Description) & " ]"
	Fn_Web_UI_ChooseFile_Operations = False
Else
	Fn_Web_UI_ChooseFile_Operations = True
End If	
'--------------------------------------------------------------------------------------------------------------------------------
'Release objects
'--------------------------------------------------------------------------------------------------------------------------------	
Set obj_AWCTeamcenterHome = Nothing
Set objWscript = Nothing
'--------------------------------------------------------------------------------------------------------------------------------	
End Function
